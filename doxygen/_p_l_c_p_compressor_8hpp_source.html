94  inline void perlocate_down(size_t pos, len_t k) {
95  auto lcp_k = (*m_lcp)[k];
96 
97  perlocation_dir_t dir = NONE;
98  do {
99  len_t lcp_lc = (lc(pos) < m_size) ? (*m_lcp)[m_heap[lc(pos)]] : 0;
100  len_t lcp_rc = (rc(pos) < m_size) ? (*m_lcp)[m_heap[rc(pos)]] : 0;
101 
102  // find perlocation direction
103  if(lcp_k < lcp_lc && lcp_k < lcp_rc) {
104  // both children are larger, pick the largest
105  dir = (lcp_lc > lcp_rc) ? LEFT : RIGHT;
106  } else if(lcp_k < lcp_lc) {
107  // go to the left
108  dir = LEFT;
109  } else if(lcp_k < lcp_rc) {
110  // go to the right
111  dir = RIGHT;
112  } else {
113  dir = NONE;
114  }
115 
116  // go down if necessary
117  if(dir == LEFT) {
118  // left
119  put(pos, m_heap[lc(pos)]);
120  pos = lc(pos);
121  } else if(dir == RIGHT) {
122  // right
123  put(pos, m_heap[rc(pos)]);
