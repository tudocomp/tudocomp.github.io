204  // construct requested structures

208

210  if(flags & PHI) { require_phi(cm_construct); release_unneeded(); }
211  if(flags & PLCP) { require_plcp(cm_construct); release_unneeded(); }
212  if(flags & LCP) { require_lcp(cm_construct); release_unneeded(); }
213  if(flags & ISA) { require_isa(cm_construct); release_unneeded(); }
214 
215  if(cm == CompressMode::delayed) {
