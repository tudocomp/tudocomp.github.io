26  // The map is defined as such:
27  // For escaping:
28  // if input byte is escape_bytes[i], then
29  // escape_bytes[i] -> [replacement_bytes[0], replacement_bytes[i]]
30  // For unescaping:
31  // if input byte is replacement_bytes[0], then
32  // [replacement_bytes[0], replacement_bytes[i]] -> escape_bytes[i]

38  inline EscapeMap(const InputRestrictions& rest) {
39  m_escape_bytes = rest.escape_bytes();
40  m_null_terminate = rest.null_terminate();
41 
42  if (!m_escape_bytes.empty()) {
43  // Determine which bytes are legal to use for escaping
44  // by using the POOL constant and removing all
45  // bytes that need escaping
46  for (uint8_t byte : ESCAPE_BYTE_POOL) {
47  bool found = false;
48  for (uint8_t x : m_escape_bytes) {
49  if (x == byte) found = true;
50  }
51  if (!found && m_replacement_bytes.size() < (m_escape_bytes.size() + 1)) {
52  m_replacement_bytes.push_back(byte);
53  }
54  }
55 
56  // m_replacement_bytes[0] is our escape byte, so add it to m_escape_bytes
57  m_escape_bytes.insert(m_escape_bytes.begin(), m_replacement_bytes.at(0));
58  }
59 
60  // For the algorithm to work we need a escape byte, and a byte for each
61  // to-be-escaped byte
62  DCHECK_EQ(m_replacement_bytes.size(), m_escape_bytes.size());
63  }
