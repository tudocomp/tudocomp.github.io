409  // Remember element offset before iterator invalidation
410  auto p = (position - cbegin());
411 
412  // Step 1: Grow backing vector by needed amount
413  {
414  auto new_bits_needed = elem2bits(n);
415  auto existing_extra_bits = backing2bits(this->m_vec.size()) - elem2bits(this->m_real_size);
416 
417  if (new_bits_needed > existing_extra_bits) {
418  new_bits_needed -= existing_extra_bits;
419  } else {
420  new_bits_needed = 0;
421  }
422 
423  auto new_backing_needed = bits2backing(new_bits_needed);
424  this->m_vec.insert(this->m_vec.cend(), new_backing_needed, 0);
425  this->m_real_size += n;
426  }
427 
428  // Step 2: move elements to back, leaving a gap
429  {
430  auto start = begin() + p;
431  auto old_end = end() - n;

441  }

445  for(auto a = begin() + p, b = begin() + p + n; a != b; ++a) {

529  auto old_width = this->width();
530  auto new_width = w;
531  auto old_size = this->size();
532  auto new_size = n;
533 
534  auto new_bit_size = elem2bits_w(new_size, new_width);
535  auto common_size = std::min(old_size, new_size);
536 
537  if (old_width < new_width) {
538  // grow
539 
540  // Read from position of last element in the old width grid,
541  // and write to position of last element in the new width grid
542  auto old_p = bitpos2backingpos_w(elem2bits_w(common_size, old_width));
543  auto new_p = bitpos2backingpos_w(elem2bits_w(common_size, new_width));
544 
545  // make room for new bits, reallocating as needed
546  this->m_vec.resize(bits2backing_w(new_bit_size));
547  this->set_width_raw(w);
548  this->m_real_size = new_size;
549 
550  uint64_t* old_ptr = this->m_vec.data() + old_p.pos;
551  uint64_t* new_ptr = this->m_vec.data() + new_p.pos;
552 
553  // move elements into new width grid
554  for (uint64_t i = 0; i < common_size; i++) {
555  sdsl::bits::move_left((const uint64_t*&) old_ptr, old_p.offset, old_width);
556  auto v = sdsl::bits::read_int( old_ptr, old_p.offset, old_width);
557 
558  sdsl::bits::move_left((const uint64_t*&) new_ptr, new_p.offset, new_width);
559  sdsl::bits::write_int( new_ptr, v, new_p.offset, new_width);
560  }
561  } else if (old_width > new_width) {
562  // shrink
563 
564  uint64_t* old_ptr = this->m_vec.data();
565  uint64_t* new_ptr = this->m_vec.data();
566 
567  uint8_t old_offset = 0;
568  uint8_t new_offset = 0;
569 
570  // move elements into new width grid
571  for (uint64_t i = 0; i < common_size; i++) {
572  auto v = sdsl::bits::read_int_and_move((const uint64_t*&) old_ptr, old_offset, old_width);
573  sdsl::bits::write_int_and_move(new_ptr, v, new_offset, new_width);
574  }
575 
576  // remove extra bits, dropping as needed
577  this->m_vec.resize(bits2backing_w(new_bit_size));
578  this->set_width_raw(w);
579  this->m_real_size = new_size;
580  }

584  auto a = this->begin() + old_size;

602  bool operator==(const BitPackingVector<N>& lhs, const BitPackingVector<N>& rhs) {
603  DCHECK(lhs.width() == rhs.width());
604  if (lhs.size() == rhs.size()) {
605  auto extra_bits = lhs.backing2bits(lhs.m_vec.size())
606  - lhs.elem2bits(lhs.m_real_size);
607 
608  if (extra_bits == 0) {
609  return lhs.m_vec == rhs.m_vec;
610  } else {
611  if (!std::equal(lhs.m_vec.cbegin(), lhs.m_vec.cend() - 1, rhs.m_vec.cbegin())) {
612  return false;
613  }
614  auto occupied_bits = lhs.backing2bits(1) - extra_bits;
615  // NB: Underflow can not happen here because there are never
616  // completely unoccupied backing integers
617  auto elements = (occupied_bits - 1) / lhs.elem2bits(1) + 1;
618 
619  return std::equal(lhs.cend() - elements, lhs.cend(), rhs.cend() - elements);
620  }
621  }
622  return false;
623  }
