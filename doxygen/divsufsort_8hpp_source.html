43 /* Sorts suffixes of type B*. */
44 template<typename buffer_t>
45 inline saidx_t sort_typeBstar(
46  const sauchar_t *T, buffer_t& SA,
47  saidx_t *bucket_A, saidx_t *bucket_B,
48  saidx_t n) {
49 
50  saidx_t PAb, ISAb, buf;
51  saidx_t i, j, k, t, m, bufsize;
52  saint_t c0, c1;
53 
54  /* Initialize bucket arrays. */
55  for(i = 0; i < BUCKET_A_SIZE; ++i) { bucket_A[i] = 0; }
56  for(i = 0; i < BUCKET_B_SIZE; ++i) { bucket_B[i] = 0; }
57 
58  /* Count the number of occurrences of the first one or two characters of each
59  type A, B and B* suffix. Moreover, store the beginning position of all
60  type B* suffixes into the array SA. */
61  for(i = n - 1, m = n, c0 = T[n - 1]; 0 <= i;) {
62  /* type A suffix. */
63  do { ++BUCKET_A(c1 = c0); } while((0 <= --i) && ((c0 = T[i]) >= c1));
64  if(0 <= i) {
65  /* type B* suffix. */
66  ++BUCKET_BSTAR(c0, c1);
67  SA[--m] = i;
68  /* type B suffix. */
69  for(--i, c1 = c0; (0 <= i) && ((c0 = T[i]) <= c1); --i, c1 = c0) {
70  ++BUCKET_B(c0, c1);
71  }
72  }
73  }
74  m = n - m;
75 /*
76 note:
77  A type B* suffix is lexicographically smaller than a type B suffix that
78  begins with the same first two characters.
79 */
80 
81  /* Calculate the index of start/end point of each bucket. */
82  for(c0 = 0, i = 0, j = 0; c0 < ALPHABET_SIZE; ++c0) {
83  t = i + BUCKET_A(c0);
84  BUCKET_A(c0) = i + j; /* start point */
85  i = t + BUCKET_B(c0, c0);
86  for(c1 = c0 + 1; c1 < ALPHABET_SIZE; ++c1) {
87  j += BUCKET_BSTAR(c0, c1);
88  BUCKET_BSTAR(c0, c1) = j; /* end point */
89  i += BUCKET_B(c0, c1);
90  }
91  }
92 
93  if(0 < m) {
94  /* Sort the type B* suffixes by their first two characters. */
95  PAb = n - m; ISAb = m;
96  for(i = m - 2; 0 <= i; --i) {
97  t = SA[PAb + i], c0 = T[t], c1 = T[t + 1];
98  SA[--BUCKET_BSTAR(c0, c1)] = i;
99  }
100  t = SA[PAb + m - 1], c0 = T[t], c1 = T[t + 1];
101  SA[--BUCKET_BSTAR(c0, c1)] = m - 1;
102 
103  /* Sort the type B* substrings using sssort. */
104  buf = m, bufsize = n - (2 * m);
105  for(c0 = ALPHABET_SIZE - 2, j = m; 0 < j; --c0) {
106  for(c1 = ALPHABET_SIZE - 1; c0 < c1; j = i, --c1) {
107  i = BUCKET_BSTAR(c0, c1);
108  if(1 < (j - i)) {
109  sssort(T, SA, PAb, i, j,
110  buf, bufsize, 2, n, SA[i] == (m - 1));
111  }
112  }
113  }
114 
115  /* Compute ranks of type B* substrings. */
116  for(i = m - 1; 0 <= i; --i) {
117  if(0 <= SA[i]) {
118  j = i;
119  do { SA[ISAb + SA[i]] = i; } while((0 <= --i) && (0 <= SA[i]));
120  SA[i + 1] = i - j;
121  if(i <= 0) { break; }
122  }
123  j = i;
124  do { SA[i] = ~SA[i]; SA[ISAb + SA[i]] = j; } while(SA[--i] < 0);
125  SA[ISAb + SA[i]] = j;
126  }
127 
128  /* Construct the inverse suffix array of type B* suffixes using trsort. */
129  trsort(SA, ISAb, 0, m, 1);
130 
131  /* Set the sorted order of tyoe B* suffixes. */
132  for(i = n - 1, j = m, c0 = T[n - 1]; 0 <= i;) {
133  for(--i, c1 = c0; (0 <= i) && ((c0 = T[i]) >= c1); --i, c1 = c0) { }
134  if(0 <= i) {
135  t = i;
136  for(--i, c1 = c0; (0 <= i) && ((c0 = T[i]) <= c1); --i, c1 = c0) { }
137  SA[SA[ISAb + (--j)]] = ((t == 0) || (1 < (t - i))) ? t : ~t;
138  }
139  }
140 
141  /* Calculate the index of start/end point of each bucket. */
142  BUCKET_B(ALPHABET_SIZE - 1, ALPHABET_SIZE - 1) = n; /* end point */
143  for(c0 = ALPHABET_SIZE - 2, k = m - 1; 0 <= c0; --c0) {
144  i = BUCKET_A(c0 + 1) - 1;
145  for(c1 = ALPHABET_SIZE - 1; c0 < c1; --c1) {
146  t = i - BUCKET_B(c0, c1);
147  BUCKET_B(c0, c1) = i; /* end point */
148 
149  /* Move all type B* suffixes to the correct position. */
150  for(i = t, j = BUCKET_BSTAR(c0, c1);
151  j <= k;
152  --i, --k) { SA[i] = SA[k]; }
153  }
154  BUCKET_BSTAR(c0, c0 + 1) = i - BUCKET_B(c0, c0) + 1; /* start point */
155  BUCKET_B(c0, c0) = i; /* end point */
156  }
157  }
158 
159  return m;
160 }

163 /* Constructs the suffix array by using the sorted order of type B* suffixes. */
164 template<typename buffer_t>
165 inline void construct_SA(
166  const sauchar_t *T, buffer_t& SA,
167  saidx_t *bucket_A, saidx_t *bucket_B,
168  saidx_t n, saidx_t m) {
169 
170  saidx_t i, j, k;
171  saidx_t s;
172  saint_t c0, c1, c2;
173 
174  if(0 < m) {
175  /* Construct the sorted order of type B suffixes by using
176  the sorted order of type B* suffixes. */
177  for(c1 = ALPHABET_SIZE - 2; 0 <= c1; --c1) {
178  /* Scan the suffix array from right to left. */
179  for(i = BUCKET_BSTAR(c1, c1 + 1),
180  j = BUCKET_A(c1 + 1) - 1, k = -1, c2 = -1;
181  i <= j;
182  --j) {
183  if(0 < (s = SA[j])) {
184  assert(T[s] == c1);
185  assert(((s + 1) < n) && (T[s] <= T[s + 1]));
186  assert(T[s - 1] <= T[s]);
187  SA[j] = ~s;
188  c0 = T[--s];
189  if((0 < s) && (T[s - 1] > c0)) { s = ~s; }
190  if(c0 != c2) {
191  if(0 <= c2) { BUCKET_B(c2, c1) = k; }
192 
193  k = BUCKET_B(c2 = c0, c1);
194  }
195  assert(k < j);
196  SA[k--] = s;
197  } else {
198  assert(((s == 0) && (T[s] == c1)) || (s < 0));
199  SA[j] = ~s;
200  }
201  }
202  }
203  }
204 
205  /* Construct the suffix array by using
206  the sorted order of type B suffixes. */
207  k = BUCKET_A(c2 = T[n - 1]);
208  SA[k++] = (T[n - 2] < c2) ? ~(n - 1) : (n - 1);
209  /* Scan the suffix array from left to right. */
210  for(i = 0, j = n; i < j; ++i) {
211  if(0 < (s = SA[i])) {
212  assert(T[s - 1] >= T[s]);
213  c0 = T[--s];
214  if((s == 0) || (T[s - 1] < c0)) { s = ~s; }
215  if(c0 != c2) {
216  BUCKET_A(c2) = k;
217  k = BUCKET_A(c2 = c0);
218  }
219  assert(i < k);
220  SA[k++] = s;
221  } else {
222  assert(s < 0);
223  SA[i] = ~s;
224  }
225  }
226 }

232  saidx_t *bucket_A, saidx_t *bucket_B, saidx_t n) {
233

262 template<typename buffer_t>
263 inline saint_t divsufsort(const sauchar_t* T, buffer_t& SA, saidx_t n) {
264  saidx_t *bucket_A, *bucket_B;
265  saidx_t m;
266  saint_t err = 0;
267 
268  /* Check arguments. */
269  if((T == NULL) || (n < 0)) { return -1; }
270  else if(n == 0) { return 0; }
271  else if(n == 1) { SA[0] = 0; return 0; }
272  else if(n == 2) { m = (T[0] < T[1]); SA[m ^ 1] = 0, SA[m] = 1; return 0; }
273 
274  bucket_A = new saidx_t[BUCKET_A_SIZE];
275  bucket_B = new saidx_t[BUCKET_B_SIZE];
276 
277  /* Suffixsort. */
278  if((bucket_A != NULL) && (bucket_B != NULL)) {
279  divsufsort_run(T, SA, bucket_A, bucket_B, n);
280  } else {
281  err = -2;
282  }
283 
284  delete[] bucket_B;
285  delete[] bucket_A;
286 
287  return err;
288 }
