501  _push_block (_size >> 8, _bheadO, ! _bheadO); // append to block Open
502  _size += 256;
503  return (_size >> 8) - 1;
504  }
505  // transfer block from one start w/ head_in to one start w/ head_out
506  void _transfer_block (const int bi, int& head_in, int& head_out) {
507  _pop_block (bi, head_in, bi == _block[bi].next);
508  _push_block (bi, head_out, ! head_out && _block[bi].num);
509  }
510  // pop empty node from block; never transfer the special block (bi = 0)
511  int _pop_enode (const int base, const uchar label, const int from) {
512  const int e = base < 0 ? _find_place () : base ^ label;
513  const int bi = e >> 8;
514  node& n = _array[e];
515  block& b = _block[bi];
516  if (--b.num == 0) {
517  if (bi) _transfer_block (bi, _bheadC, _bheadF); // Closed to Full
518  } else { // release empty node from empty ring
519  _array[-n.base_].check = n.check;
520  _array[-n.check].base_ = n.base_;
521  if (e == b.ehead) b.ehead = -n.check; // set ehead
522  if (bi && b.num == 1 && b.trial != MAX_TRIAL) // Open to Closed
523  _transfer_block (bi, _bheadO, _bheadC);
524  }
525  // initialize the released node
526 #ifdef USE_REDUCED_TRIE
527  n.value = CEDAR_VALUE_LIMIT; n.check = from;
528  if (base < 0) _array[from].base_ = - (e ^ label) - 1;
529 #else
530  if (label) n.base_ = -1; else n.value = value_type (0); n.check = from;
531  if (base < 0) _array[from].base_ = e ^ label;
532 #endif
533  return e;
534  }
535  // push empty node into empty ring
536  void _push_enode (const int e) {
537  const int bi = e >> 8;
538  block& b = _block[bi];
539  if (++b.num == 1) { // Full to Closed
540  b.ehead = e;
541  _array[e] = node (-e, -e);
542  if (bi) _transfer_block (bi, _bheadF, _bheadC); // Full to Closed
543  } else {
544  const int prev = b.ehead;
545  const int next = -_array[prev].check;
546  _array[e] = node (-prev, -next);
547  _array[prev].check = _array[next].base_ = -e;
548  if (b.num == 2 || b.trial == MAX_TRIAL) // Closed to Open
549  if (bi) _transfer_block (bi, _bheadC, _bheadO);
550  b.trial = 0;
551  }
552  if (b.reject < _reject[b.num]) b.reject = _reject[b.num];
553  _ninfo[e] = ninfo (); // reset ninfo; no child, no sibling
554  }
555  // push label to from's child
556  void _push_sibling (const size_t from, const int base, const uchar label, const bool flag = true) {
557  uchar* c = &_ninfo[from].child;
558  if (flag && (ORDERED ? label > *c : ! *c))
559  do c = &_ninfo[base ^ *c].sibling; while (ORDERED && *c && *c < label);
560  _ninfo[base ^ label].sibling = *c, *c = label;
561  }
562  // pop label from from's child
563  void _pop_sibling (const size_t from, const int base, const uchar label) {
564  uchar* c = &_ninfo[from].child;
565  while (*c != label) c = &_ninfo[base ^ *c].sibling;
566  *c = _ninfo[base ^ label].sibling;
567  }
568  // check whether to replace branching w/ the newly added node
569  bool _consult (const int base_n, const int base_p, uchar c_n, uchar c_p) const {
570  do c_n = _ninfo[base_n ^ c_n].sibling, c_p = _ninfo[base_p ^ c_p].sibling;
571  while (c_n && c_p);
572  return c_p;
573  }
574  // enumerate (equal to or more than one) child nodes
575  uchar* _set_child (uchar* p, const int base, uchar c, const int label = -1) {
576  --p;
577  if (! c) { *++p = c; c = _ninfo[base ^ c].sibling; } // 0: terminal
578  if (ORDERED)
579  while (c && c < label) { *++p = c; c = _ninfo[base ^ c].sibling; }
580  if (label != -1) *++p = static_cast <uchar> (label);
581  while (c) { *++p = c; c = _ninfo[base ^ c].sibling; }
582  return p;
583  }
584  // explore new block to settle down
585  int _find_place () {
586  if (_bheadC) return _block[_bheadC].ehead;
587  if (_bheadO) return _block[_bheadO].ehead;
588  return _add_block () << 8;
589  }
590  int _find_place (const uchar* const first, const uchar* const last) {
591  if (int bi = _bheadO) {
592  const int bz = _block[_bheadO].prev;
593  const short nc = static_cast <short> (last - first + 1);
594  while (1) { // set candidate block
595  block& b = _block[bi];
596  if (b.num >= nc && nc < b.reject) // explore configuration
597  for (int e = b.ehead;;) {
598  const int base = e ^ *first;
599  for (const uchar* p = first; _array[base ^ *++p].check < 0; )
600  if (p == last) return b.ehead = e; // no conflict
601  if ((e = -_array[e].check) == b.ehead) break;
602  }
603  b.reject = nc;
604  if (b.reject < _reject[b.num]) _reject[b.num] = b.reject;
605  const int bi_ = b.next;
606  if (++b.trial == MAX_TRIAL) _transfer_block (bi, _bheadO, _bheadC);
607  if (bi == bz) break;
608  bi = bi_;
609  };
610  }
611  return _add_block () << 8;
612  }
613  // resolve conflict on base_n ^ label_n = base_p ^ label_p
614  template <typename T>
615  int _resolve (size_t& from_n, const int base_n, const uchar label_n, T& cf) {
616  // examine siblings of conflicted nodes
617  const int to_pn = base_n ^ label_n;
618  const int from_p = _array[to_pn].check;
619  const int base_p = _array[from_p].base ();
620  const bool flag // whether to replace siblings of newly added
621  = _consult (base_n, base_p, _ninfo[from_n].child, _ninfo[from_p].child);
622  uchar child[256];
623  uchar* const first = &child[0];
624  uchar* const last =
625  flag ? _set_child (first, base_n, _ninfo[from_n].child, label_n)
626  : _set_child (first, base_p, _ninfo[from_p].child);
627  const int base =
628  (first == last ? _find_place () : _find_place (first, last)) ^ *first;
629  // replace & modify empty list
630  const int from = flag ? static_cast <int> (from_n) : from_p;
631  const int base_ = flag ? base_n : base_p;
632  if (flag && *first == label_n) _ninfo[from].child = label_n; // new child
633 #ifdef USE_REDUCED_TRIE
634  _array[from].base_ = -base - 1; // new base
635 #else
636  _array[from].base_ = base; // new base
637 #endif
638  for (const uchar* p = first; p <= last; ++p) { // to_ => to
639  const int to = _pop_enode (base, *p, from);
640  const int to_ = base_ ^ *p;
641  _ninfo[to].sibling = (p == last ? 0 : *(p + 1));
642  if (flag && to_ == to_pn) continue; // skip newcomer (no child)
643  cf (to_, to); // user-defined callback function to handle moved nodes
644  node& n = _array[to];
645  node& n_ = _array[to_];
646 #ifdef USE_REDUCED_TRIE
647  if ((n.base_ = n_.base_) < 0 && *p) // copy base; bug fix
648 #else
649  if ((n.base_ = n_.base_) > 0 && *p) // copy base; bug fix
650 #endif
651  {
652  uchar c = _ninfo[to].child = _ninfo[to_].child;
653  do _array[n.base () ^ c].check = to; // adjust grand son's check
654  while ((c = _ninfo[n.base () ^ c].sibling));
655  }
656  if (! flag && to_ == static_cast <int> (from_n)) // parent node moved
657  from_n = static_cast <size_t> (to); // bug fix
658  if (! flag && to_ == to_pn) { // the address is immediately used
659  _push_sibling (from_n, to_pn ^ label_n, label_n);
660  _ninfo[to_].child = 0; // remember to reset child
661 #ifdef USE_REDUCED_TRIE
662  n_.value = CEDAR_VALUE_LIMIT;
663 #else
664  if (label_n) n_.base_ = -1; else n_.value = value_type (0);
665 #endif
666  n_.check = static_cast <int> (from_n);
667  } else
668  _push_enode (to_);
669  if (NUM_TRACKING_NODES) // keep the traversed node updated
670  for (size_t j = 0; tracking_node[j] != 0; ++j)
671  if (tracking_node[j] == static_cast <size_t> (to_))
672  { tracking_node[j] = static_cast <size_t> (to); break; }
673  }
674  return flag ? base ^ label_n : to_pn;
675  }
676  };
