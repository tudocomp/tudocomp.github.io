39 /* Simple insertionsort for small size groups. */
40 template<typename buffer_t>
41 inline void tr_insertionsort(buffer_t& B, const saidx_t ISAd, saidx_t first, saidx_t last) {
42  saidx_t a, b;
43  saidx_t t, r;
44 
45  for(a = first + 1; a < last; ++a) {
46  for(t = B[a], b = a - 1; 0 > (r = B[ISAd + t] - B[ISAd + B[b]]);) {
47  do { B[b + 1] = B[b]; } while((first <= --b) && (B[b] < 0));
48  if(b < first) { break; }
49  }
50  if(r == 0) { B[b] = ~B[b]; }
51  B[b + 1] = t;
52  }
53 }

59 inline void tr_fixdown(buffer_t& B, const saidx_t ISAd, saidx_t SA, saidx_t i, saidx_t size) {

73 /* Simple top-down heapsort. */
74 template<typename buffer_t>
75 inline void tr_heapsort(buffer_t& B, const saidx_t ISAd, saidx_t SA, saidx_t size) {
76  saidx_t i, m;
77  saidx_t t;
78 
79  m = size;
80  if((size % 2) == 0) {
81  m--;
82  if(B[ISAd + B[SA + m / 2]] < B[ISAd + B[SA + m]]) { SWAP(B[SA + m], B[SA + m / 2]); }
83  }
84 
85  for(i = m / 2 - 1; 0 <= i; --i) { tr_fixdown(B, ISAd, SA, i, m); }
86  if((size % 2) == 0) { SWAP(B[SA + 0], B[SA + m]); tr_fixdown(B, ISAd, SA, 0, m); }
87  for(i = m - 1; 0 < i; --i) {
88  t = B[SA + 0], B[SA + 0] = B[SA + i];
89  tr_fixdown(B, ISAd, SA, 0, i);
90  B[SA + i] = t;
91  }

104  else { return v3; }

110 template<typename buffer_t>

123 /* Returns the pivot element. */
124 template<typename buffer_t>
125 inline saidx_t tr_pivot(buffer_t& B, const saidx_t ISAd, saidx_t first, saidx_t last) {
126  saidx_t middle;
127  saidx_t t;
128 
129  t = last - first;
130  middle = first + t / 2;
131 
132  if(t <= 512) {
133  if(t <= 32) {
134  return tr_median3(B, ISAd, first, middle, last - 1);
135  } else {
136  t >>= 2;
137  return tr_median5(B, ISAd, first, first + t, middle, last - 1 - t, last - 1);
138  }
139  }
140  t >>= 3;
141  first = tr_median3(B, ISAd, first, first + t, first + (t << 1));
142  middle = tr_median3(B, ISAd, middle - t, middle, middle + t);
143  last = tr_median3(B, ISAd, last - 1 - (t << 1), last - 1 - t, last - 1);
144  return tr_median3(B, ISAd, first, middle, last);
145 }

174 template<typename buffer_t>
175 inline void tr_partition(buffer_t& B, const saidx_t ISAd,
176  saidx_t first, saidx_t middle, saidx_t last,
177  saidx_t *pa, saidx_t *pb, saidx_t v) {
178  saidx_t a, b, c, d, e, f;
179  saidx_t t, s;
180  saidx_t x = 0;
181 
182  for(b = middle - 1; (++b < last) && ((x = B[ISAd + B[b]]) == v);) { }
183  if(((a = b) < last) && (x < v)) {
184  for(; (++b < last) && ((x = B[ISAd + B[b]]) <= v);) {
185  if(x == v) { SWAP(B[b], B[a]); ++a; }
186  }
187  }
188  for(c = last; (b < --c) && ((x = B[ISAd + B[c]]) == v);) { }
189  if((b < (d = c)) && (x > v)) {
190  for(; (b < --c) && ((x = B[ISAd + B[c]]) >= v);) {
191  if(x == v) { SWAP(B[c], B[d]); --d; }
192  }
193  }
194  for(; b < c;) {
195  SWAP(B[b], B[c]);
196  for(; (++b < c) && ((x = B[ISAd + B[b]]) <= v);) {
197  if(x == v) { SWAP(B[b], B[a]); ++a; }
198  }
199  for(; (b < --c) && ((x = B[ISAd + B[c]]) >= v);) {
200  if(x == v) { SWAP(B[c], B[d]); --d; }
201  }
202  }
203 
204  if(a <= d) {
205  c = b - 1;
206  if((s = a - first) > (t = b - a)) { s = t; }
207  for(e = first, f = b - s; 0 < s; --s, ++e, ++f) { SWAP(B[e], B[f]); }
208  if((s = d - c) > (t = last - d - 1)) { s = t; }
209  for(e = b, f = last - s; 0 < s; --s, ++e, ++f) { SWAP(B[e], B[f]); }
210  first += (b - a), last -= (d - c);
211  }
212  *pa = first, *pb = last;
213 }

503 /* Tandem repeat sort */
504 template<typename buffer_t>
505 inline void trsort(buffer_t& B, saidx_t ISA, saidx_t SA, saidx_t n, saidx_t depth) {
506  saidx_t ISAd;
507  saidx_t first, last;
508  trbudget_t budget;
509  saidx_t t, skip, unsorted;
510 
511  trbudget_init(&budget, ilg<saidx_t>(n) * 2 / 3, n);
512  for(ISAd = ISA + depth; -n < B[SA]; ISAd += ISAd - ISA) {
513  first = SA;
514  skip = 0;
515  unsorted = 0;
516  do {
517  if((t = B[first]) < 0) { first -= t; skip += t; }
518  else {
519  if(skip != 0) { B[first + skip] = skip; skip = 0; }
520  last = SA + B[ISA + t] + 1;
521  if(1 < (last - first)) {
522  budget.count = 0;
523  tr_introsort(B, ISA, ISAd, SA, first, last, &budget);
524  if(budget.count != 0) { unsorted += budget.count; }
525  else { skip = first - last; }
526  } else if((last - first) == 1) {
527  skip = -1;
528  }
529  first = last;
530  }
531  } while(first < (SA + n));
532  if(skip != 0) { B[first + skip] = skip; }
533  if(unsorted == 0) { break; }
534  }
535 }
