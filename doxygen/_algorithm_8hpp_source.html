296  Meta&& meta)
297 {
298  // get vector for the current type
299  auto& target_type_algos = target[meta.type()];
300 
301  // Create decl::Algorithm value here
302 
303  auto decl_value = std::move(meta).build_def();
304 
305  bool found = false;
306  for (auto& already_seen_algo : target_type_algos) {
307  if (already_seen_algo.name() == decl_value.name()) {
308  if (already_seen_algo != decl_value) {
309  std::stringstream ss;
310  ss << "Attempting to register the same algorithm twice";
311  ss << " with differing details:";
312  ss << " new: " << decl_value;
313  ss << " existing: " << already_seen_algo;
314  throw std::runtime_error(ss.str());
315  }
316 
317  found = true;
318  break;
319  }
320  }
321 
322  if (!found) {
323  target_type_algos.push_back(std::move(decl_value));
324  }

327 }
