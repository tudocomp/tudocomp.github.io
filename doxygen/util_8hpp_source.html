456 /*
457  * Square root by abacus algorithm, Martin Guy @ UKC, June 1985.
458  * From a book on programming abaci by Mr C. Woo.
459  * and from https://en.wikipedia.org/wiki/Methods_of_computing_square_roots
460  */
461 template<class int_t>
462 int_t isqrt(int_t num) {
463  int_t res = 0;
464  int_t bit = 1ULL << (sizeof(int_t)-2); // The second-to-top bit is set: 1 << 30 for 32 bits
465 
466  // "bit" starts at the highest power of four <= the argument.
467  while (bit > num)
468  bit >>= 2;
469 
470  while (bit != 0) {
471  if (num >= res + bit) {
472  num -= res + bit;
473  res = (res >> 1) + bit;
474  }
475  else
476  res >>= 1;
477  bit >>= 2;
478  }
479  return res;
480 }
