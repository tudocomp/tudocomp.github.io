62 ss_isqrt(saidx_t x) {
63  saidx_t y, e;
64 
65  if(x >= (SS_BLOCKSIZE * SS_BLOCKSIZE)) { return SS_BLOCKSIZE; }
66  e = (x & 0xffff0000) ?
67  ((x & 0xff000000) ?
68  24 + lg_table[(x >> 24) & 0xff] :
69  16 + lg_table[(x >> 16) & 0xff]) :
70  ((x & 0x0000ff00) ?
71  8 + lg_table[(x >> 8) & 0xff] :
72  0 + lg_table[(x >> 0) & 0xff]);
73 
74  if(e >= 16) {
75  y = sqq_table[x >> ((e - 6) - (e & 1))] << ((e >> 1) - 7);
76  if(e >= 24) { y = (y + 1 + x / y) >> 1; }
77  y = (y + 1 + x / y) >> 1;
78  } else if(e >= 8) {
79  y = (sqq_table[x >> ((e - 6) - (e & 1))] >> (7 - (e >> 1))) + 1;
80  } else {
81  return sqq_table[x] >> 4;
82  }
83 
84  return (x < (y * y)) ? y - 1 : y;

93 template<typename buffer_t1, typename buffer_t2>
94 inline saint_t ss_compare(const sauchar_t *T,
95  buffer_t1& B1, saidx_t p1,
96  buffer_t2& B2, saidx_t p2,
97  saidx_t depth) {
98  const sauchar_t *U1, *U2, *U1n, *U2n;
99 
100  for(U1 = T + depth + B1[p1],
101  U2 = T + depth + B2[p2],
102  U1n = T + B1[p1 + 1] + 2,
103  U2n = T + B2[p2 + 1] + 2;
104  (U1 < U1n) && (U2 < U2n) && (*U1 == *U2);
105  ++U1, ++U2) {
106  }
107 
108  return U1 < U1n ?
109  (U2 < U2n ? *U1 - *U2 : 1) :
110  (U2 < U2n ? -1 : 0);
111 }

211 /* Returns the pivot element. */
212 template<typename buffer_t>
213 inline saidx_t ss_pivot(const sauchar_t *Td, buffer_t& B, const saidx_t PA, saidx_t first, saidx_t last) {
214  saidx_t middle;
215  saidx_t t;
216 
217  t = last - first;
218  middle = first + t / 2;
219 
220  if(t <= 512) {
221  if(t <= 32) {
222  return ss_median3(Td, B, PA, first, middle, last - 1);
223  } else {
224  t >>= 2;
225  return ss_median5(Td, B, PA, first, first + t, middle, last - 1 - t, last - 1);
226  }
227  }
228  t >>= 3;
229  first = ss_median3(Td, B, PA, first, first + t, first + (t << 1));
230  middle = ss_median3(Td, B, PA, middle - t, middle, middle + t);
231  last = ss_median3(Td, B, PA, last - 1 - (t << 1), last - 1 - t, last - 1);
232  return ss_median3(Td, B, PA, first, middle, last);
233 }

238 /* Binary partition for substrings. */
239 template<typename buffer_t>
240 inline saidx_t ss_partition(buffer_t& B, const saidx_t PA,
241  saidx_t first, saidx_t last, saidx_t depth) {
242  saidx_t a, b;
243  saidx_t t;
244  for(a = first - 1, b = last;;) {
245  for(; (++a < b) && ((B[PA + B[a]] + depth) >= (B[PA + B[a] + 1] + 1));) { B[a] = ~B[a]; }
246  for(; (a < --b) && ((B[PA + B[b]] + depth) < (B[PA + B[b] + 1] + 1));) { }
247  if(b <= a) { break; }
248  t = ~B[b];
249  B[b] = B[a];
250  B[a] = t;
251  }
252  if(first < a) { B[first] = ~B[first]; }
253  return a;
254 }

273 #if 1 < SS_INSERTIONSORT_THRESHOLD

275 #endif
276  STACK_POP(first, last, depth, limit);
277  continue;
278  }
279 
280  Td = T + depth;
281  if(limit-- == 0) {
282  ss_heapsort(Td, B, PA, first, last - first);
283  }
284  if(limit < 0) {
285  for(a = first + 1, v = Td[B[PA + B[first]]]; a < last; ++a) {
286  if((x = Td[B[PA + B[a]]]) != v) {
287  if(1 < (a - first)) { break; }
288  v = x;
289  first = a;
290  }
291  }
292  if(Td[B[PA + B[first]] - 1] < v) {
293  first = ss_partition(B, PA, first, a, depth);
294  }
295  if((a - first) <= (last - a)) {
296  if(1 < (a - first)) {
297  STACK_PUSH(a, last, depth, -1);
298  last = a, depth += 1, limit = ilg<saidx_t>(a - first);
299  } else {
300  first = a, limit = -1;
301  }
302  } else {
303  if(1 < (last - a)) {
304  STACK_PUSH(first, a, depth + 1, ilg<saidx_t>(a - first));
305  first = a, limit = -1;
306  } else {
307  last = a, depth += 1, limit = ilg<saidx_t>(a - first);
308  }
309  }

311  }
312 
313  /* choose pivot */
314  a = ss_pivot(Td, B, PA, first, last);

327  for(; (b < --c) && ((x = Td[B[PA + B[c]]]) >= v);) {
328  if(x == v) { SWAP(B[c], B[d]); --d; }
329  }
330  }
331  for(; b < c;) {
332  SWAP(B[b], B[c]);
333  for(; (++b < c) && ((x = Td[B[PA + B[b]]]) <= v);) {
334  if(x == v) { SWAP(B[b], B[a]); ++a; }
335  }
336  for(; (b < --c) && ((x = Td[B[PA + B[c]]]) >= v);) {
337  if(x == v) { SWAP(B[c], B[d]); --d; }
338  }
339  }
340 
341  if(a <= d) {
342  c = b - 1;
343 
344  if((s = a - first) > (t = b - a)) { s = t; }
345  for(e = first, f = b - s; 0 < s; --s, ++e, ++f) { SWAP(B[e], B[f]); }
346  if((s = d - c) > (t = last - d - 1)) { s = t; }

350  b = (v <= Td[B[PA + B[a]] - 1]) ? a : ss_partition(B, PA, a, c, depth);

352  if((a - first) <= (last - c)) {
353  if((last - c) <= (c - b)) {
354  STACK_PUSH(b, c, depth + 1, ilg<saidx_t>(c - b));
355  STACK_PUSH(c, last, depth, limit);
356  last = a;
357  } else if((a - first) <= (c - b)) {
358  STACK_PUSH(c, last, depth, limit);

363  STACK_PUSH(first, a, depth, limit);

366  } else {

371  } else if((last - c) <= (c - b)) {
372  STACK_PUSH(first, a, depth, limit);
