91  size_t A[2*alphabet_size];
92  for(size_t i=0; i < alphabet_size; i++) {
93  DVLOG(2) << "Char " << map_from_effective[i] << " : " << size_t(C[map_from_effective[i]]);

95  A[i] = alphabet_size+i;
96  }
97  // build a minHeap of A[0..size)
98  auto comp = [&A] (const size_t a, const size_t b) -> bool { return A[a] > A[b];};
99  std::make_heap(&A[0], &A[alphabet_size], comp);
100  DCHECK_LE(A[A[0]], A[A[1]]);
101  assert_permutation_offset<size_t*>(A,alphabet_size,alphabet_size);
102 
103  DVLOG(2) << "A: "<<arr_to_debug_string(A,2*alphabet_size);
104 
105 
106  size_t h = alphabet_size-1;
107  //builds the Huffman tree
108  while(h > 0) {
109  std::pop_heap(A, A+h+1, comp);
110  const size_t m1 = A[h]; // first element from the heap (already popped)
111  --h;
112 
113  std::pop_heap(A, A+h+1, comp);
114  const size_t m2 = A[h]; //second min count
115  DCHECK_GT(m1,h); DCHECK_GT(m2,h);
116 
117  A[h+1] = A[m1] + A[m2]; // create a new parent node
118  A[h] = h + 1;
119  A[m1] = A[m2] = h + 1; //parent pointer
120  std::push_heap(A, A+h+1, comp);
121  }
122 
123  A[1] = 0;
124  for(size_t i = 2; i < 2*alphabet_size; ++i) {
125  A[i] = A[A[i]]+ 1;
126  }
127  // variant:
128  // for(size_t i = size; i < 2*size; ++i) {
129  // size_t d = 0;
130  // size_t r = i;
131  // while(r > 1) {
132  // ++d;
133  // r = A[r];
134  // }
135  // A[i] = d;
136  // }

139  for (size_t i=0; i < alphabet_size; i++) {
140  DCHECK_LE(A[alphabet_size+i], 64); // the latter representation allows only codewords of length at most 64 bits
141  codelengths[i] = A[alphabet_size+i];
142  DVLOG(2) << "Char " << map_from_effective[i] << " : " << codelengths[i];
143  }
144 
145  IF_PARANOID({
146  // invariants
147 
148  // check that more frequent keywords get shorter codelengthss
149  for(size_t i=0; i < alphabet_size; ++i) {
150  for(size_t j=i+1; j < alphabet_size; ++j) {
151  if(codelengths[i] > codelengths[j]) {
152  DCHECK_LE(C[map_from_effective[i]], C[map_from_effective[j]]);
153  }
154  else if(codelengths[i] < codelengths[j]) {
155  DCHECK_GE(C[map_from_effective[i]], C[map_from_effective[j]]);
156  }
157  }}

159  const size_t& max_el = *std::max_element(A+alphabet_size,A+2*alphabet_size);
160  DCHECK_LT(max_el,63);
161  size_t sum = 0;

163  {
164  sum += 2ULL<<(max_el - A[alphabet_size+i]);
165  }
166  DCHECK_EQ(sum, 2ULL<<max_el);
167  }
168  })

445  DCHECK_GT(alphabet_size,0);

455  std::iota(codeword_order,codeword_order+alphabet_size,0);

457

459

461  uint8_t* ordered_codelengths { new uint8_t[alphabet_size] };
462  uliteral_t*const ordered_map_from_effective { new uliteral_t[alphabet_size] };
463  for(size_t i = 0; i < alphabet_size; ++i) {
464  ordered_codelengths[i] = codelengths[codeword_order[i]];
465  ordered_map_from_effective[i] = mapFromEffective[codeword_order[i]];
